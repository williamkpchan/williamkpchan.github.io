A grouped Rose chart

This goes in the documents header: 
<script src="RGraph.svg.common.core.js"></script>
<script src="RGraph.svg.common.tooltips.js"></script>
<script src="RGraph.svg.rose.js"></script>
Put this where you want the chart to show up: 
<div id="cc" style="width: 1000px; height: 700px"></div>
This is the code that generates the chart: 
<script>
    new RGraph.SVG.Rose({
        id: 'cc',
        data: [4,6,3,5,2,8,9],
        options: {
            margin: .3,
            segmentsAngleOffset: -.15,
            labels: ['Monday','Tuesday','Wednesday','Thursday','Friday','Saturday','Sunday'],
            tooltips: [
                'Richard on Monday: 4%',
                'Richard on Tuesday: 6%',
                'Richard on Wednesday: 3%',
                'Richard on Thursday: 5%',
                'Richard on Friday: 2%',
                'Richard on Saturday: 8%',
                'Richard on Sunday: 9%'
            ],
            labelsSize: 28
        }
    }).draw();

    new RGraph.SVG.Rose({
        id: 'cc',
        data: [5,8,6,3,2,9,9],
        options: {
            colors: ['pink'],
            margin: .3,
            segmentsAngleOffset: .15,
            tooltips: [
                'Jane on Monday: 5%',
                'Jane on Tuesday: 8%',
                'Jane on Wednesday: 6%',
                'Jane on Thursday: 3%',
                'Jane on Friday: 2%',
                'Jane on Saturday: 9%',
                'Jane on Sunday: 9%'
            ],
        }
    }).draw();
    
    RGraph.SVG.tooltips.style.color           = 'white';
    RGraph.SVG.tooltips.style.backgroundColor = 'black';
    RGraph.SVG.tooltips.style.fontWeight      = 'bold';
    RGraph.SVG.tooltips.style.fontSize        = '20pt';
    RGraph.SVG.tooltips.style.padding         = '15px';
</script>
