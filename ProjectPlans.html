  shiftedge = function(edge) {
    return(
      c(tail(edge, length(edge)-1), sample(-10:10, 1))
    )
  }

# calcWave
  calcWAve = function(theArray) {
    sum = 0
    for( i in 1:length(theArray) ) {sum = sum + theArray[i] * i;}
    return (sum / ((1 + length(theArray))*length(theArray)/2))
  }
